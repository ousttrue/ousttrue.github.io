import neovim # msgpack-rpc で vim と通信する


@neovim.plugin
class RemotePlugin:

    def __init__(self, nvim):
        self.nvim = nvim

    @neovim.command('HogeCommand') # vimのコマンドとして実行可能になる

        pass
