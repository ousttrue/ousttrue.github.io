CMAKE_MINIMUM_REQUIRED(VERSION 3.0.0)
PROJECT(hello VERSION 0.1.0)

FILE(GLOB SRC
    *.cpp
    *.h

ADD_EXECUTABLE(${PROJECT_NAME} ${SRC})
